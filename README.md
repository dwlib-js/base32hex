# Base32Hex

## Install
`npm i --save @dwlib/base32hex`

## Usage
```javascript
// CJS
const Base32Hex = require('@dwlib/base32hex');
const Base32HexDecode = require('@dwlib/base32hex/decode');
const Base32HexDecodeBigInt = require('@dwlib/base32hex/decodeBigInt');
const Base32HexDecodeInt = require('@dwlib/base32hex/decodeInt');
const Base32HexDecodeInto = require('@dwlib/base32hex/decodeInto');
const Base32HexEncode = require('@dwlib/base32hex/encode');
const Base32HexEncodeBigInt = require('@dwlib/base32hex/encodeBigInt');
const Base32HexEncodeInt = require('@dwlib/base32hex/encodeInt');
const Base32HexIsValid = require('@dwlib/base32hex/isValid');
const Base32HexNormalize = require('@dwlib/base32hex/normalize');
const Base32HexValidate = require('@dwlib/base32hex/validate');
// ESM
import Base32Hex from '@dwlib/base32hex';
import Base32HexDecode from '@dwlib/base32hex/decode';
import Base32HexDecodeBigInt from '@dwlib/base32hex/decodeBigInt';
import Base32HexDecodeInt from '@dwlib/base32hex/decodeInt';
import Base32HexDecodeInto from '@dwlib/base32hex/decodeInto';
import Base32HexEncode from '@dwlib/base32hex/encode';
import Base32HexEncodeBigInt from '@dwlib/base32hex/encodeBigInt';
import Base32HexEncodeInt from '@dwlib/base32hex/encodeInt';
import Base32HexIsValid from '@dwlib/base32hex/isValid';
import Base32HexNormalize from '@dwlib/base32hex/normalize';
import Base32HexValidate from '@dwlib/base32hex/validate';
```
